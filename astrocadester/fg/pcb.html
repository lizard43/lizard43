Copy/paste pattern

The renderer script must be loaded before the inline customization call.

<link rel="stylesheet" href="css/astrocade-pcb.css" />
<script src="js/astrocade-pcb.js"></script>

<div id="my-board" class="astrocade-pcb-host"></div>
<script>
  AstrocadePCB.renderBoard(document.querySelector('#my-board'), {
    title: 'Custom Game Board',
    width: 1000,
    height: 680,
    edgeConnector: { fingers: 11 },
    pcbLabels: [
      { position: 'top-left', text: 'A082-91356', color: 'green' }
    ],
    components: [
      { type: 'ic', ref: 'U1', label: 'Z80 CPU', x: 420, y: 245, w: 145, h: 78, pins: 40 },
      { type: 'resistor', ref: 'R1', x: 420, y: 360, w: 94 }
    ]
  });
</script>
